// GAS - 双方向リアルタイムランキング通信・完全大開通版 const SHEET_NAME = "GameLogs"; function doPost(e) { // どんな想定外のエラーが起きてもCORSを維持し、確実にJSONオブジェクトをフロントに返す try { const data = JSON.parse(e.postData.contents); const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME) || SpreadsheetApp.getActiveSpreadsheet().insertSheet(SHEET_NAME); // ヘッダー行の自動生成 if (sheet.getLastRow() === 0) { sheet.appendRow([ "timestamp", "session_id", "event_type", "play_count", "load_time_ms", "camera_permission", "play_duration_sec", "total_score", "hit_taps", "miss_taps", "tap_accuracy", "email", "sns_click_wacca", "sns_click_snafu", "nickname" ]); } // 1. スプレッドシートへのデータ追加（H列=スコア, O列=名前） const now = new Date(); sheet.appendRow([ now.toISOString(), String(data.session_id || ""), String(data.event_type || "popup"), Number(data.play_count) || 1, Number(data.load_time_ms) || 0, String(data.camera_permission || "granted"), Number(data.play_duration_sec) || 10, Number(data.total_score) || 0, // H列(8列目) Number(data.hit_taps) || 0, Number(data.miss_taps) || 0, Number(data.tap_accuracy) || 0, String(data.email || ""), data.sns_click_wacca || false, data.sns_click_snafu || false, String(data.nickname || "ANON").trim().toUpperCase() // O列(15列目) ]); // セルのデータ追加を強制確定 SpreadsheetApp.flush(); const lastRow = sheet.getLastRow(); const allPlayers = []; // 2. スプレッドシートのH列とO列からデータを一括スキャン if (lastRow > 1) { const rawValues = sheet.getRange(2, 1, lastRow - 1, 15).getValues(); for (let i = 0; i < rawValues.length; i++) { const row = rawValues[i]; const scoreVal = Number(row[7]); // H列 (total_score) let nameVal = String(row[14]).trim(); // O列 (nickname) if (!nameVal || nameVal === "undefined" || nameVal === "") { nameVal = "ANON"; } if (!isNaN(scoreVal)) { allPlayers.push({ nickname: nameVal.toUpperCase(), score: scoreVal }); } } } // 3. スコア降順（高い順）にソートして上位15名を厳密に抽出 allPlayers.sort(function(a, b) { return b.score - a.score; }); const top15 = allPlayers.slice(0, 15); // 相対順位（パーセンタイル）計算 const myScore = Number(data.total_score) || 0; const betterCount = allPlayers.filter(function(p) { return p.score > myScore; }).length; const totalCount = allPlayers.length || 1; const percentile = Math.min(100, Math.max(0, Math.round((betterCount / totalCount) * 100))); // 4. フロントへ返すJSONレスポンスのビルド const resPayload = { success: true, percentile: percentile, total_players: totalCount, leaderboard: top15 }; // 💡 Googleの内部セキュリティを貫通させる完全なTextOutputオブジェクトへの格納 return ContentService.createTextOutput(JSON.stringify(resPayload)) .setMimeType(ContentService.MimeType.JSON) .setHeader("Access-Control-Allow-Origin", "*") .setHeader("Access-Control-Allow-Methods", "POST, GET, OPTIONS") .setHeader("Access-Control-Allow-Headers", "Content-Type"); } catch (err) { // 例外エラー時もCORSを維持して、フロント側のコンソールにエラー原因を直接返す const errPayload = { success: false, error: err.toString(), leaderboard: [] }; return ContentService.createTextOutput(JSON.stringify(errPayload)) .setMimeType(ContentService.MimeType.JSON) .setHeader("Access-Control-Allow-Origin", "*"); } } function doGet(e) { return ContentService.createTextOutput(JSON.stringify({ status: "ok" })) .setMimeType(ContentService.MimeType.JSON) .setHeader("Access-Control-Allow-Origin", "*"); }
